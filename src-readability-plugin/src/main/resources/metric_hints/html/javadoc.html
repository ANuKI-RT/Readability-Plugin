Good Practice

Here a well written javadoc is present.


/**
* Create a chatroom for 2 users of the chat service.
* In order to authenticate against the api the authtoken of the
* first chat user is used, due to the fact that this user initated creating the chat room.
*
* @param firstMember the first member of the chatroom (this one create the room)
* @param secondMember the second member of the chatroom
* @param title the title of the chatroom (if null the displaynames of the members are used as default name)
* @param description the description of the chatroom (if null a empty string will be set)
* @return the created chat room
* @throws ApiException if the api request to send the request failed
*/
public ChatRoom createChatRoom(User firstMember, User secondMember, String title, String description) throws ApiException {
    ChatApiService apiService = ChatApiService.getInstance();
    RoomApi roomApi = apiService.createRoomApi(firstMember.authToken());

    if(title == null) {
        title = "Chat Room between " firstMember.getUserName() + " and " + secondMember.getUserName();
    }

    if(description == null) {
        description = "";
    }

    return roomApi.createChatRoom(firstMember.getId(),secondMember.getId(),title,description);
}
